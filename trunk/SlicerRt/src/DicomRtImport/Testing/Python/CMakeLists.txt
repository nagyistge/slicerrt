foreach(extension ${extensions})
  SlicerMacroConfigureGenericPythonModuleTests("${extension}" KIT_UNITTEST_SCRIPTS)

    SCRIPT ${script_name}
    SLICER_ARGS --no-main-window TESTNAME_PREFIX nomainwindow_)
endforeach()
